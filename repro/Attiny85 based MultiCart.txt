NesResetDetector.ino
-------------------------------------------------------------
#include "Arduino.h"

#define LED 1 //P1
#define M2_PIN 1 //P2
#define S0 3 //P3

int m2Reading = 0;
long resetTime = -1;

void setup() {
	pinMode(S0, OUTPUT);
	digitalWrite(S0, LOW);

	pinMode(LED, OUTPUT);
	digitalWrite(LED, HIGH);
}

void loop() {
  m2Reading = analogRead(M2_PIN); //Read P2

  if (m2Reading > 100 && resetTime != -1) {
	  resetTime = -1;
  }

  if (m2Reading < 100) {
	  if (resetTime == -1)
		  resetTime = millis();
	  else if (resetTime >=0 ) {
		  long resetDuration = millis() - resetTime;
		  if (resetDuration > 500) {
			  digitalWrite(S0,  HIGH);
			  delay(1);
			  digitalWrite(S0,  LOW);
			  digitalWrite(LED, !digitalRead(LED));
			  resetTime = -2;
		  }
	  }
  }
}
